rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // User documents and nested subcollections
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Explicitly allow all nested documents under users
      match /{document=**} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }
    
    // Lootboxes collection - split read permissions for clarity
    match /lootboxes/{lootboxId} {
      // Allow reading individual documents if user owns them
      allow get: if request.auth != null 
                 && resource.data.uid == request.auth.uid;
      
      // Allow listing/querying the collection for authenticated users
      allow list: if request.auth != null;
      
      // Allow create if user is authenticated and setting their own uid
      allow create: if request.auth != null 
                    && request.auth.uid == request.resource.data.uid;
      
      // Allow update/delete if user owns it
      allow update, delete: if request.auth != null 
                            && request.auth.uid == resource.data.uid;
    }
    
    // Group boxes
    match /group_boxes/{groupBoxId} {
      // Anyone can read group boxes (for sharing)
      allow read: if true;
      
      // Create: only if you're the creator
      allow create: if request.auth != null
                    && request.auth.uid == request.resource.data.createdBy;
      
      // Update: allow authenticated users
      allow update: if request.auth != null;
      
      // Delete: only creator
      allow delete: if request.auth != null
                    && request.auth.uid == resource.data.createdBy;
      
      // Sub-collection: opens (anyone can read/create opens)
      match /opens/{openId} {
        allow read: if true;
        allow create: if request.auth != null;
        allow update, delete: if false;
      }
      
      // Sub-collection: user tries tracking
      match /user_tries/{userId} {
        allow read: if true;
        allow write: if request.auth != null;
      }
      
      // Sub-collection: session history
      match /session_history/{historyId} {
        allow read: if true;
        allow create: if request.auth != null;
        allow update, delete: if false;
      }
    }
    
    // Test collection
    match /tests/{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Spins collection
    match /spins/{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Chests collection
    match /chests/{document=**} {
      allow read: if true;
    }
  }
}
